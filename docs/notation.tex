\begin{notation}

  \begin{notationlist}{3em}
    \item[$S$] 全量 Skill 集合
    \item[$s_i$] 第 $i$ 个 Skill
    \item[$\mathbf{d}_i$] Skill $s_i$ 的 Description embedding
    \item[$\mathbf{v}_i$] Skill $s_i$ 的 Content embedding
    \item[$\mathcal{N}(q)$] 查询 $q$ 对应的语义邻域
    \item[$K$] K-NN 图的邻居数或 Top-$K$ 推荐数量
    \item[$d$] PageRank 风格迭代中的阻尼因子
    \item[$SR(s_i)$] Skill $s_i$ 的 SkillRank 分值
    \item[$\rho$] within-cluster Spearman 秩相关系数
  \end{notationlist}

\end{notation}
